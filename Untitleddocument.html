<script>

const input = document.getElementById("imageInput");

const btn = document.getElementById("generateBtn");

btn.addEventListener("click", () => {

    if(!input.files[0]){

        alert("Upload image first");

        return;

    }

    const img = new Image();

    img.onload = () => {

        if(document.getElementById("height").checked){

            generateHeightMap(img);

        }

        if(document.getElementById("normal").checked){

            generateNormalMap(img);

        }

        if(document.getElementById("ao").checked){

            generateAOMap(img);

        }

    };

    img.src = URL.createObjectURL(input.files[0]);

});

function createCanvas(img){

    const canvas = document.createElement("canvas");

    canvas.width = img.width;

    canvas.height = img.height;

    const ctx = canvas.getContext("2d");

    ctx.drawImage(img,0,0);

    return {canvas,ctx};

}

function download(canvas,name){

    const a = document.createElement("a");

    a.href = canvas.toDataURL("image/png");

    a.download = name;

    a.click();

}

function generateHeightMap(img){

    const {canvas,ctx} = createCanvas(img);

    const data = ctx.getImageData(

        0,0,

        canvas.width,

        canvas.height

    );

    for(let i=0;i<data.data.length;i+=4){

        const gray =

        (

            data.data[i] +

            data.data[i+1] +

            data.data[i+2]

        ) / 3;

        data.data[i] = gray;

        data.data[i+1] = gray;

        data.data[i+2] = gray;

    }

    ctx.putImageData(data,0,0);

    download(canvas,"HeightMap.png");

}

function generateAOMap(img){

    const {canvas,ctx} = createCanvas(img);

    const data = ctx.getImageData(

        0,0,

        canvas.width,

        canvas.height

    );

    for(let i=0;i<data.data.length;i+=4){

        let gray =

        (

            data.data[i] +

            data.data[i+1] +

            data.data[i+2]

        ) / 3;

        gray *= 0.7;

        data.data[i] = gray;

        data.data[i+1] = gray;

        data.data[i+2] = gray;

    }

    ctx.putImageData(data,0,0);

    download(canvas,"AOMap.png");

}

function generateNormalMap(img){

    const {canvas,ctx} = createCanvas(img);

    const data = ctx.getImageData(

        0,0,

        canvas.width,

        canvas.height

    );

    for(let i=0;i<data.data.length;i+=4){

        data.data[i] = 128;

        data.data[i+1] = 128;

        data.data[i+2] = 255;

    }

    ctx.putImageData(data,0,0);

    download(canvas,"NormalMap.png");

}
